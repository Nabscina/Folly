1   package follybot.gui;

3   import follybot.follybot.Conversation;
4   import java.awt.Color;
5   import java.awt.event.ActionEvent;
6   import java.awt.event.ActionListener;
7   import javax.swing.JFrame;
8   import javax.swing.JTextField;
9   
10  /**
11   * Tapahtumankuuntelija, eli kun tapahtuma tapahtuu (painetaan enteriä),
12   * actionPerformed päivittää tekstikentät + voi myös sulkea ikkunan Follyn
13   * tekstikentän perusteella. Myös Follyn fontin värin asetus käy kätevästi
14   * samalla kun se vaihtuu vasta kun käyttäjän nimi on tiedossa.
15   */
16  public class ActionListening implements ActionListener {
17  
18      private JTextField follysfield;
19      private JTextField humansfield;
20      private JFrame frame;
21  
22      private Conversation convo;
23  
24      /**
25       * Asetetaan parametreinä saadut frame, folly, human ja convo oliomuuttujien
26       * arvoiksi, palauttaa ActionListening-olion.
27       *
28       * @param frame GUI-luokan JFrame-olio.
29       * @param folly Follyn tekstikenttä.
30       * @param human käyttäjän tekstikenttä.
31       * @param convo Conversation-olio, jolta saadaan Follyn tekstikenttään
32       * asetettava vastaus.
33       */
34      public ActionListening(JFrame frame, JTextField folly, JTextField human, Conversation convo) {
35  
36          this.follysfield = folly;
37          this.humansfield = human;
38          this.frame = frame;
39  
40          this.convo = convo;
41      }
42  
43      /**
44       * Metodi päivittää Follyn tekstikenttään vastauksen käyttäjälle ja
45       * tyhjentää käyttäjän tekstikentän uutta syötettä varten, sekä myös muuttaa
46       * Follyn tekstin väriä kun metodia ekan kerran kutsutaan (nimen antamisen
47       * jälkeen) ja tekee Follyn tekstikentälle tarkistuksen, jonka perusteella
48       * voidaan sulkea ikkuna ja lopettaa ohjelma. Metodista lähdetään vitun
49       * äkkiä pois jos käyttäjä syöttää tyhjää (eikä olla hyvästelty ensin), eli
50       * tyhjä syöte ei saa vastausta, eikä esim. tyhjän nimen syöttäminen
51       * onnistu.
52       *
53       * @param ae tällainen ActionEvent generoidaan, kun enteriä painetaan. Emme
54       * nyt kyllä tee sillä mitään.
55       *
56       */
57      @Override
58      public void actionPerformed(ActionEvent ae) {
59  
60          if (humansfield.getText().trim().equals("") && !windowDisposeCheck(follysfield.getText())) {
61              humansfield.setText("");
62              return;
63          }
64  
65          if (windowDisposeCheck(follysfield.getText())) {
66              frame.dispose();
67          }
68  
69          follysfield.setForeground(Color.decode("#FF004F"));
70          follysfield.setText(convo.converse(humansfield.getText()));
71          humansfield.setText("");
72      }
73  
74      /**
75       * Tarkistetaan, lukeeko Follyn tekstikentässä equalsin jälkeinen lause (jos
76       * lukee, ikkuna suljetaan).
77       *
78       * @param follysline teksti Follyn tekstikentässä.
79       *
80       * @return true jos tekstikentän sisältö on tuo lause, muuten false.
81       */
82      public boolean windowDisposeCheck(String follysline) {
83  
84          return follysline.equals("Bye. Press enter to leave.");
85      }
86  }
