71      public String quote() {
72  
73          if (quotes1.isEmpty() || quotes2.isEmpty()) {
74              addQuotes();
75          }
76  
77          Collections.shuffle(quotes1);
78          Collections.shuffle(quotes2);
79  
80          return cl.codeToNormal(quotes1.get(0)) + cl.codeToNormal(quotes2.get(0));
81      }
