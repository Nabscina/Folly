1   package follybot.response;

3   import follybot.response.math.BotMath;
4   import java.util.Random;
5   
6   public class ResponseLogic {
7   
8       private ResponseBank rb;
9       private BotMath math;
10      private Random random;
11  
12      public ResponseLogic() {
13  
14          this.rb = new ResponseBank();
15          this.math = new BotMath();
16          this.random = new Random();
17      }
18  
19      public String respond(String question) {
20  
21          if (simplified(question).equals("bye")) {
22              return "Bye.";
23          } else if (math.mathCheck(question)) {
24              return math.doMath(question).toString();
25          }
26  
27          return rb.quote();
28      }
29  
30      public String newName() {
31  
32          return rb.getAName();
33      }
34  
35      public boolean needsChanging(String name) {
36  
37          return simplified(name).equals("folly");
38      }
39  
40      public String introduction(String name) {
41  
42          if (needsChanging(name)) {
43              String newName = newName();
44              return respondToEqualName(newName);
45          } else {
46              return respondToUniqueName(name);
47          }
48      }
49  
50      public String respondToEqualName(String name) {
51  
52          rb.setHumanName(name);
53          return rb.equalNameResponse();
54      }
55  
56      public String respondToUniqueName(String name) {
57  
58          rb.setHumanName(name);
59          return rb.uniqueNameResponse();
60      }
61  
62      public String simplified(String question) {
63  
64          return question.replaceAll("[^a-zA-Z]", "").toLowerCase();
65      }
66  }
