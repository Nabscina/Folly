1   package follybot.response.quotemaker;
2   
3   import java.util.ArrayList;
4   import java.util.Random;
5   
6   /**
7    * ResponseBank hyödyntää tätä luokkaa elämänviisauksien rakentamiseen. Täällä
8    * tarkastellaan viisauksia ja etsitään sieltä Å-merkkiä, joka tarkoittaa, että
9    * merkin voi korvata satunnaisella randomWords-listan sanalla. Jos merkkiä ei
10   * löydy, palautetaan käsitelty lause sellaisenaan. (paitsi että saatetaan myös
11   * lisätä elämänviisauden loppuun käyttäjän nimi.)
12   */
13  public class QuoteMaker {
14  
15      private Random random;
16  
17      /**
18       * Alustaa uuden Random-olion, palauttaa QuoteMaker-olion.
19       */
20      public QuoteMaker() {
21  
22          random = new Random();
23      }
24  
25      /**
26       * Tarkistaa, sisältääkö quote-merkkijono merkkiä Å, jolloin se korvataan
27       * satunnaisella sanalla listasta randomWords.
28       *
29       * @param quote ResponseBank-luokassa muodostettu lause.
30       * @param randomWords ResponseBankin lista sanoja, joista jollain voidaan
31       * tarvittaessa korvata Å-merkki quote-lauseessa.
32       * @param name käyttäjän nimi.
33       *
34       * @return merkkijono, jossa on jotain muuta Å-merkin tilalla jos Å-merkki
35       * löytyi tai alkuperäinen quote, jos Å-merkkiä ei löytynyt + lopetus
36       * lauseelle metodilla endQuote.
37       */
38      public String makeAQuote(String quote, ArrayList<String> randomWords, String name) {
39  
40          String newQuote = quote;
41  
42          if (quote.contains("Å")) {
43              newQuote = quote.replaceAll("Å", randomWords.get(0));
44          }
45  
46          return newQuote + endQuote(name);
47      }
48  
49      /**
50       * Generoi satunnaisen kokonaisluvun välillä 0-7, ja luvun perusteella
51       * päätetään, millainen lopetus palautetaan metodille makeAQuote. Käyttäjän
52       * nimen sisältävällä lopetuksella on siis 1/8 chanssi, ajattelin sen olevan
53       * sopiva.
54       *
55       * @param name käyttäjän nimi.
56       *
57       * @return pistelopetus, jos zerotoseven on muu kuin 7, muuten ", name.".
58       */
59      public String endQuote(String name) {
60  
61          int zerotoseven = random.nextInt(8);
62  
63          if (zerotoseven == 7) {
64              return ", " + name + ".";
65          } else {
66              return ".";
67          }
68      }
69  }
