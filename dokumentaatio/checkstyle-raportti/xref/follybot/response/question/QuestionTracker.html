1   package follybot.response.question;
2   
3   import java.util.HashMap;
4   
5   /**
6    * Luokka pitää kirjaa käyttäjän kysymistä kyllä tai ei -kysymyksistä ja Follyn
7    * vastauksista niihin. Tarkoitus on, että ensimmäinen vastaus tallennetaan
8    * kysymys-avaimen arvoksi, ja kun sama kysymys kysytään uudestaan, palautetaan
9    * aina se yksi alkuperäinen vastaus.
10   */
11  public class QuestionTracker {
12  
13      private HashMap<String, String> answers;
14  
15      /**
16       * Alustaa HashMap-olion talletettaville kysymyksille ja vastauksille,
17       * palauttaa QuestionTracker-olion.
18       */
19      public QuestionTracker() {
20  
21          answers = new HashMap<>();
22      }
23  
24      /**
25       * Lisää mappiin avaimeksi käyttäjän kysymyksen, jos se ei jo ole siellä.
26       * Arvoksi Follyn ResponseBank-lähtöinen vastaus. Muunnetaan kysymyksen
27       * merkit pieniksi, jotta myöhemmin sama vastaus voidaan antaa, vaikka
28       * avaimen ja kysymyksen merkkien koolla olisikin eroa.
29       *
30       * @param question käyttäjän syöte, tässä tapauksessa kysymys.
31       * @param answer Follyn vastaus, jonka BotQuestion välittää
32       * ResponseBankilta.
33       */
34      public void addAnswer(String question, String answer) {
35  
36          if (!answers.containsKey(question.toLowerCase())) {
37              answers.put(question.toLowerCase(), answer);
38          }
39      }
40  
41      /**
42       * Tarkistaa, että answers sisältää avaimena käyttäjän pienikirjaimisen
43       * kysymyksen (vaikka BotQuestion huolehtiikin aina sen lisäyksestä) ja
44       * etsii sen arvon.
45       *
46       * @param question käyttäjän syöte, nyt kysymys.
47       *
48       * @return pienimerkkisen question-kysymyksen arvo answers-mapista jos (kun)
49       * se sieltä löytyy, muuten null (mutta ei se palauta null, dw).
50       */
51      public String getAnswer(String question) {
52  
53          if (answers.containsKey(question.toLowerCase())) {
54              return answers.get(question.toLowerCase());
55          }
56  
57          return null;
58      }
59      
60      public HashMap<String, String> getAnswersMap() {
61          
62          return answers;
63      }
64  
65  }
